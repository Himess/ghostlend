## 2. FULL `IERC7984` interface — VERBATIM

Source: `_repos/openzeppelin-confidential-contracts/contracts/interfaces/IERC7984.sol` (complete file):

```solidity
// SPDX-License-Identifier: MIT
// OpenZeppelin Confidential Contracts (last updated v0.3.0) (interfaces/IERC7984.sol)
pragma solidity >=0.8.24;

import {euint64, externalEuint64} from "@fhevm/solidity/lib/FHE.sol";
import {IERC165} from "@openzeppelin/contracts/interfaces/IERC165.sol";

/// @dev Draft interface for a confidential fungible token standard utilizing the Zama FHE library.
interface IERC7984 is IERC165 {
    /**
     * @dev Emitted when the expiration timestamp for an operator `operator` is updated for a given `holder`.
     * The operator may move any amount of tokens on behalf of the holder until the timestamp `until`.
     */
    event OperatorSet(address indexed holder, address indexed operator, uint48 until);

    /// @dev Emitted when a confidential transfer is made from `from` to `to` of encrypted amount `amount`.
    event ConfidentialTransfer(address indexed from, address indexed to, euint64 indexed amount);

    /**
     * @dev Emitted when an encrypted amount is disclosed.
     *
     * Accounts with access to the encrypted amount `encryptedAmount` that is also accessible to this contract
     * should be able to disclose the amount. This functionality is implementation specific.
     */
    event AmountDisclosed(euint64 indexed encryptedAmount, uint64 amount);

    /// @dev Returns the name of the token.
    function name() external view returns (string memory);

    /// @dev Returns the symbol of the token.
    function symbol() external view returns (string memory);

    /// @dev Returns the number of decimals of the token. Recommended to be 6.
    function decimals() external view returns (uint8);

    /// @dev Returns the contract URI. See https://eips.ethereum.org/EIPS/eip-7572[ERC-7572] for details.
    function contractURI() external view returns (string memory);

    /// @dev Returns the confidential total supply of the token.
    function confidentialTotalSupply() external view returns (euint64);

    /// @dev Returns the confidential balance of the account `account`.
    function confidentialBalanceOf(address account) external view returns (euint64);

    /// @dev Returns true if `spender` is currently an operator for `holder`.
    function isOperator(address holder, address spender) external view returns (bool);

    /**
     * @dev Sets `operator` as an operator for `holder` until the timestamp `until`.
     *
     * NOTE: An operator may transfer any amount of tokens on behalf of a holder while approved.
     */
    function setOperator(address operator, uint48 until) external;

    /**
     * @dev Transfers the encrypted amount `encryptedAmount` to `to` with the given input proof `inputProof`.
     *
     * Returns the encrypted amount that was actually transferred.
     */
    function confidentialTransfer(
        address to,
        externalEuint64 encryptedAmount,
        bytes calldata inputProof
    ) external returns (euint64);

    /**
     * @dev Similar to {confidentialTransfer-address-externalEuint64-bytes} but without an input proof. The caller
     * *must* already be allowed by ACL for the given `amount`.
     */
    function confidentialTransfer(address to, euint64 amount) external returns (euint64 transferred);

    /**
     * @dev Transfers the encrypted amount `encryptedAmount` from `from` to `to` with the given input proof
     * `inputProof`. `msg.sender` must be either `from` or an operator for `from`.
     *
     * Returns the encrypted amount that was actually transferred.
     */
    function confidentialTransferFrom(
        address from,
        address to,
        externalEuint64 encryptedAmount,
        bytes calldata inputProof
    ) external returns (euint64);

    /**
     * @dev Similar to {confidentialTransferFrom-address-address-externalEuint64-bytes} but without an input proof.
     * The caller *must* be already allowed by ACL for the given `amount`.
     */
    function confidentialTransferFrom(address from, address to, euint64 amount) external returns (euint64 transferred);

    /**
     * @dev Similar to {confidentialTransfer-address-externalEuint64-bytes} but with a callback to `to` after
     * the transfer.
     *
     * The callback is made to the {IERC7984Receiver-onConfidentialTransferReceived} function on the
     * to address with the actual transferred amount (may differ from the given `encryptedAmount`) and the given
     * data `data`.
     */
    function confidentialTransferAndCall(
        address to,
        externalEuint64 encryptedAmount,
        bytes calldata inputProof,
        bytes calldata data
    ) external returns (euint64 transferred);

    /// @dev Similar to {confidentialTransfer-address-euint64} but with a callback to `to` after the transfer.
    function confidentialTransferAndCall(
        address to,
        euint64 amount,
        bytes calldata data
    ) external returns (euint64 transferred);

    /**
     * @dev Similar to {confidentialTransferFrom-address-address-externalEuint64-bytes} but with a callback to `to`
     * after the transfer.
     */
    function confidentialTransferFromAndCall(
        address from,
        address to,
        externalEuint64 encryptedAmount,
        bytes calldata inputProof,
        bytes calldata data
    ) external returns (euint64 transferred);

    /**
     * @dev Similar to {confidentialTransferFrom-address-address-euint64} but with a callback to `to`
     * after the transfer.
     *
     */
    function confidentialTransferFromAndCall(
        address from,
        address to,
        euint64 amount,
        bytes calldata data
    ) external returns (euint64 transferred);
}
```

## 3. `ERC7984` base implementation (v0.5.1)

Source: `_repos/openzeppelin-confidential-contracts/contracts/token/ERC7984/ERC7984.sol`. `abstract contract ERC7984 is IERC7984, ERC165`. Storage: `mapping(address holder => euint64) _balances; mapping(address holder => mapping(address spender => uint48)) _operators; euint64 _totalSupply;`.

Constructor: `constructor(string memory name_, string memory symbol_, string memory contractURI_)`.

### 3.2 Internal API

- `function _setOperator(address holder, address operator, uint48 until) internal virtual` — writes `_operators[holder][operator] = until`, emits `OperatorSet`.
- `function _mint(address to, euint64 amount) internal returns (euint64 transferred)` — `require(to != address(0), ERC7984InvalidReceiver(address(0))); return _update(address(0), to, amount);`
- `function _burn(address from, euint64 amount) internal returns (euint64 transferred)` — `require(from != address(0), ERC7984InvalidSender(address(0))); return _update(from, address(0), amount);`
- `function _transfer(address from, address to, euint64 amount) internal returns (euint64 transferred)` — both zero-address checks, then `_update`.
- `function _update(address from, address to, euint64 amount) internal virtual returns (euint64 transferred)` — THE central hook (verbatim, full body):

```solidity
function _update(address from, address to, euint64 amount) internal virtual returns (euint64 transferred) {
    ebool success;
    euint64 ptr;

    if (from == address(0)) {
        (success, ptr) = FHESafeMath.tryIncrease(_totalSupply, amount);
        FHE.allowThis(ptr);
        _totalSupply = ptr;
    } else {
        euint64 fromBalance = _balances[from];
        (success, ptr) = FHESafeMath.tryDecrease(fromBalance, amount);
        FHE.allowThis(ptr);
        FHE.allow(ptr, from);
        _balances[from] = ptr;
    }

    transferred = FHE.select(success, amount, FHE.asEuint64(0));

    if (to == address(0)) {
        ptr = FHE.sub(_totalSupply, transferred);
        FHE.allowThis(ptr);
        _totalSupply = ptr;
    } else {
        ptr = FHE.add(_balances[to], transferred);
        FHE.allowThis(ptr);
        FHE.allow(ptr, to);
        _balances[to] = ptr;
    }

    if (from != address(0)) FHE.allow(transferred, from);
    if (to != address(0)) FHE.allow(transferred, to);
    FHE.allowThis(transferred);
    emit ConfidentialTransfer(from, to, transferred);
}
```

Natspec verbatim: "Safely moves up to `amount` from `from` to `to`, or mints/burns if `from`/`to` is the zero address. Emits a {ConfidentialTransfer} event with the successfully transferred amount."

### 3.3 Insufficient-balance behavior — CLAMP TO ZERO, ALL-OR-NOTHING (drives lending accounting)

- `FHESafeMath.tryDecrease` (verbatim, `contracts/utils/FHESafeMath.sol`): `success = FHE.ge(oldValue, delta); updated = FHE.select(success, FHE.sub(oldValue, delta), oldValue);` — and in `_update`: `transferred = FHE.select(success, amount, FHE.asEuint64(0));`
- **Consequence:** if the sender's encrypted balance is `>= amount`, the FULL amount moves; otherwise the transaction still SUCCEEDS but moves an encrypted **0**. There is **no partial fill and no revert** on insufficient balance. The returned `euint64 transferred` handle is the *actual* amount moved (either `amount` or `0`) — a lending pool MUST credit the depositor with `transferred`, never with the requested amount.
- README warning (verbatim): "many functions do not revert on failure as they would in normal contracts."
- CHANGELOG 0.5.0 (verbatim): "`ERC7984`: Remove revert on transfer where the sender has an uninitialized balance. ([#357])" — i.e., since v0.5.0 the library no longer reverts even for accounts that never held tokens.
- **Deployed-wrapper divergence (Zama's on-chain ConfidentialWrapper, incl. Sepolia cToken mocks):** `_raw/erc7984/confidential-wrapper.md` documents error `ERC7984ZeroBalance(holder)` — "Sender has never held tokens" — and warns (verbatim): "Accounts with a zero balance that have never held tokens cannot be the `from` address in confidential transfers." (Same restriction stated for unwrap requests.) So the *deployed* wrapper reverts for never-initialized senders while library v0.5.1 clamps silently. **UNCERTAIN:** which exact library version the deployed mocks are built on (?ask could not answer — `_raw/erc7984/ask-mocks-operator.md`).
- ?ask response on clamping (`_raw/erc7984/ask-clamping.md`, verbatim excerpt): "the on-chain transfer logic is designed to **avoid reverts and instead conditionally transfer zero** ... So when `canTransfer` is false (insufficient encrypted balance), the **net transferred amount is the encrypted value of `0`**." (SDK adds a client-side pre-flight `InsufficientConfidentialBalanceError` before sending, bypassable with `skipBalanceCheck: true`.)
- Mint path overflow: `tryIncrease` clamps too — on total-supply overflow, mint succeeds but mints 0 (the ERC-20 wrapper adds a hard revert, section 6).

### 3.4 ACL semantics of the returned `transferred` handle

From `_update` + `confidentialTransferFrom` code above:
- `from` gets **persistent** `FHE.allow(transferred, from)`;
- `to` gets **persistent** `FHE.allow(transferred, to)`;
- the token contract itself gets `FHE.allowThis(transferred)`;
- the **operator (msg.sender) gets only `FHE.allowTransient(transferred, msg.sender)`** — valid for the current transaction only.
- Practical consequence for a pool: when the pool calls `confidentialTransferFrom(user, address(pool), ...)`, the pool is `to`, so it holds a **persistent** ACL allowance on `transferred` and may use it in the same tx (e.g., `FHE.add` into its accounting) — but any NEW handle the pool computes must itself get `FHE.allowThis(...)` to be usable in later transactions. ?ask response (`_raw/erc7984/ask-acl-returned-handle.md`): "transient allowances ... are valid for the duration of the current transaction... For a lending pool ... grant permanent ACL permission (typically via `FHE.allowThis(...)`)".
- Errors (verbatim names, base contract): `ERC7984InvalidReceiver(address receiver)`, `ERC7984InvalidSender(address sender)`, `ERC7984UnauthorizedSpender(address holder, address spender)`, `ERC7984UnauthorizedUseOfEncryptedAmount(euint64 amount, address user)`, `ERC7984UnauthorizedCaller(address caller)`; extra event `AmountDiscloseRequested(euint64 indexed encryptedAmount, address indexed requester)`.

## 4. OPERATOR flow — how a pool pulls user funds

### 4.1 Semantics

- Holder-side grant (verbatim): `function setOperator(address operator, uint48 until) external;` — `until` is a unix timestamp in **seconds**, `uint48`. Active while `block.timestamp <= until` (inclusive; see `isOperator` body in 3.1). Revoke by calling `setOperator(op, 0)` (any past timestamp). No per-amount allowance exists — it is all-or-nothing while active.
- Repo docs (`docs/modules/ROOT/pages/token.adoc`, verbatim): "If Bob is an operator for Alice, Bob can move any amount of Alice's tokens at any point in time. Operators are set using an expiration timestamp--this can be thought of as a limited duration infinite approval for an `ERC20`."
  - "NOTE: Operators do not have allowance to reencrypt/decrypt balance handles for other addresses. This means that operators cannot transfer full balances and can only know success after a transaction (by decrypting the transferred amount)."
  - "WARNING: Setting an operator for any amount of time allows the operator to _**take all of your tokens**_. Carefully vet all potential operators before giving operator approval."
- SDK default: `token.setOperator("0xOperator")` uses a **1-hour** default expiry; pass unix-seconds timestamp for custom windows (`_raw/erc7984/operator-approvals.md`, `sdk-token.md`).

### 4.2 Step-by-step: pool pulls a deposit via `confidentialTransferFrom`

1. **User → token:** `token.setOperator(poolAddress, until)` (holder tx; `until` = unix seconds, e.g. now+86400).
2. **User (off-chain):** build encrypted input bound to the **contract that will call `FHE.fromExternal`** and to the user: `fhevm.createEncryptedInput(poolAddress, user.address).add64(amount).encrypt()` → `{ handles[0]: externalEuint64, inputProof }`. (If the user instead calls the token's `externalEuint64` overload directly, bind to the token address — see doc test in `_raw/erc7984/erc7984.md` which uses `createEncryptedInput(await token.getAddress(), owner.address)`.)
3. **User → pool:** `pool.deposit(externalAmount, inputProof)`.
4. **Pool internal:** `euint64 amount = FHE.fromExternal(externalAmount, inputProof);` then **`FHE.allowTransient(amount, address(token));`** (the token must be ACL-allowed to use the pool-created ciphertext), then `euint64 transferred = token.confidentialTransferFrom(msg.sender, address(this), amount);`.
5. **Token:** checks `isOperator(user, pool)`; clamps to 0 if balance insufficient; credits pool balance; grants pool persistent ACL on `transferred` (pool is `to`) + transient (pool is caller).
6. **Pool accounting:** `deposits[user] = FHE.add(deposits[user], transferred); FHE.allowThis(deposits[user]); FHE.allow(deposits[user], user);` — always use `transferred`, never the requested amount.

### 4.3 Verbatim example — `SwapERC7984ToERC7984` (repo `contracts/mocks/docs/SwapERC7984ToERC7984.sol`; same file embedded in OZ docs)

```solidity
contract SwapERC7984ToERC7984 {
    function swapConfidentialForConfidential(
        IERC7984 fromToken,
        IERC7984 toToken,
        externalEuint64 amountInput,
        bytes calldata inputProof
    ) public virtual {
        require(fromToken.isOperator(msg.sender, address(this)));

        euint64 amount = FHE.fromExternal(amountInput, inputProof);

        FHE.allowTransient(amount, address(fromToken));
        euint64 amountTransferred = fromToken.confidentialTransferFrom(msg.sender, address(this), amount);

        FHE.allowTransient(amountTransferred, address(toToken));
        toToken.confidentialTransfer(msg.sender, amountTransferred);
    }
}
```

Repo docs annotate the steps (verbatim): "1. Check operator approval 2. Allow the `fromToken` to access `amount` 3. Transfer from `from` to this contract for `amount` 4. Allow the `toToken` to access `amountTransferred` 5. Transfer `amountTransferred` to `msg.sender`".

### 4.4 Verbatim example — `SwapERC7984ToERC20` (repo `contracts/mocks/docs/SwapERC7984ToERC20.sol`; identical in `_raw/erc7984/swaperc7984toerc20.md`) — also shows the public-decryption finalize pattern a pool needs for withdrawals to ERC-20

```solidity
contract SwapERC7984ToERC20 {
    error SwapERC7984ToERC20InvalidFinalization(euint64 amount);

    mapping(euint64 amount => address) private _receivers;
    IERC7984 private _fromToken;
    IERC20 private _toToken;

    constructor(IERC7984 fromToken, IERC20 toToken) {
        _fromToken = fromToken;
        _toToken = toToken;
    }

    function swapConfidentialToERC20(externalEuint64 encryptedInput, bytes memory inputProof) public {
        euint64 amount = FHE.fromExternal(encryptedInput, inputProof);
        FHE.allowTransient(amount, address(_fromToken));
        euint64 amountTransferred = _fromToken.confidentialTransferFrom(msg.sender, address(this), amount);

        FHE.makePubliclyDecryptable(amountTransferred);
        _receivers[amountTransferred] = msg.sender;
    }

    function finalizeSwap(euint64 amount, uint64 cleartextAmount, bytes calldata decryptionProof) public virtual {
        bytes32[] memory handles = new bytes32[](1);
        handles[0] = euint64.unwrap(amount);

        FHE.checkSignatures(handles, abi.encode(cleartextAmount), decryptionProof);
        address to = _receivers[amount];
        require(to != address(0), SwapERC7984ToERC20InvalidFinalization(amount));
        delete _receivers[amount];

        if (cleartextAmount != 0) {
            SafeERC20.safeTransfer(_toToken, to, cleartextAmount);
        }
    }
}
```

Note: `swapConfidentialToERC20` relies on the caller having previously done `setOperator(swapContract, until)` on `_fromToken` (the `confidentialTransferFrom` operator check).

## 5. Transfer callbacks — `onConfidentialTransferReceived`

`IERC7984Receiver` (verbatim, complete, `contracts/interfaces/IERC7984Receiver.sol`):

```solidity
/// @dev Interface for contracts that can receive ERC7984 transfers with a callback.
interface IERC7984Receiver {
    /**
     * @dev Called upon receiving a confidential token transfer. Returns an encrypted boolean indicating success
     * of the callback. If false is returned, the token contract will attempt to refund the transfer.
     *
     * NOTE: The calling contract (token) must be granted ACL allowance to read the confidential return value.
     *
     * WARNING: Do not manually refund the transfer AND return false, as this can lead to double refunds.
     */
    function onConfidentialTransferReceived(
        address operator,
        address from,
        euint64 amount,
        bytes calldata data
    ) external returns (ebool);
}
```

Mechanism (`ERC7984Utils.checkOnTransferReceived` + `ERC7984._transferAndCall`, verbatim semantics):
- Callback fires only if `to.code.length > 0` (EOAs: treated as accepted, `FHE.asEbool(true)`). A contract recipient that does NOT implement the interface causes revert `ERC7984InvalidReceiver(to)` (empty-reason catch) or bubbles the receiver's revert reason.
- Ordering in `_transferAndCall`: (1) `sent = _transfer(from, to, amount)` (balances updated first), (2) callback with the **actual** `sent` amount, (3) refund attempt `refund = _update(to, from, FHE.select(success, 0, sent))`, (4) `transferred = FHE.sub(sent, refund); FHE.allowTransient(transferred, msg.sender);`.
- Natspec warnings (verbatim): "The refund triggered when {IERC7984Receiver-onConfidentialTransferReceived} returns an encrypted false is best-effort only. A receiver that transfers, burns, or otherwise reduces its balance during the hook can still return false, in which case the refund transfers zero tokens. The sender's tokens end up with the recipient rather than being refunded." And: "Refunds are subject to the same validation flow as a normal transfer--they may fail for a variety of reasons ... In these cases, the tokens do not return to the sender."
- "The returned `transferred` amount is a fresh ciphertext computed as `sent - refund` and `msg.sender` only receives a transient FHE allowance for it. This value is generally intended to be processed only in the same transaction. Event observers see `sent` and `refund` individually." (verbatim natspec)
- Repo docs (`token.adoc`, verbatim): "callbacks allow the operator approval step to be skipped and directly invoke the receiver contract via a callback." → a pool can accept deposits via `confidentialTransferAndCall` + implementing `IERC7984Receiver` (user needs NO setOperator), doing its accounting inside `onConfidentialTransferReceived(operator, from, amount, data)` and returning `FHE.asEbool(true)`. The token must be able to read the returned ebool (grant ACL on it, e.g. `FHE.allowTransient(result, msg.sender)` inside the receiver — receiver's caller is the token).

## 6. ERC-20 wrapper — `ERC7984ERC20Wrapper`

Source: `contracts/token/ERC7984/extensions/ERC7984ERC20Wrapper.sol` + `contracts/interfaces/IERC7984ERC20Wrapper.sol`; Zama deployed-wrapper doc `_raw/erc7984/confidential-wrapper.md`.

Contract natspec WARNING (verbatim): "Minting assumes the full amount of the underlying token transfer has been received, hence some non-standard tokens such as fee-on-transfer or other deflationary-type tokens are not supported by this wrapper."

### 6.1 Signatures (verbatim)

- `function wrap(address to, uint256 amount) public virtual override returns (euint64)` — "Tokens are exchanged at a fixed rate specified by {rate} such that `amount / rate()` confidential tokens are sent. The amount transferred in is rounded down to the nearest multiple of {rate}." Body: `safeTransferFrom(underlying, msg.sender, address(this), amount - (amount % rate()))`, then `_mint(to, FHE.asEuint64(SafeCast.toUint64(amount / rate())))`; returns the minted handle with `FHE.allowTransient(wrappedAmountSent, msg.sender)`. Requires prior ERC-20 `approve` of the wrapper.
- `function onTransferReceived(address /*operator*/, address from, uint256 amount, bytes calldata data) public virtual returns (bytes4)` — ERC-1363 single-tx wrap path; wraps to `address(bytes20(data))` or `from`; **refunds excess** `amount % rate()` back to `from`; only callable by the underlying token.
- `function unwrap(address from, address to, euint64 amount) public virtual returns (bytes32)` — no-proof variant; requires `FHE.isAllowed(amount, msg.sender)`.
- `function unwrap(address from, address to, externalEuint64 encryptedAmount, bytes calldata inputProof) public virtual returns (bytes32)` — "The unwrap request created by this function must be finalized by calling {finalizeUnwrap}."
- `function finalizeUnwrap(bytes32 unwrapRequestId, uint64 unwrapAmountCleartext, bytes calldata decryptionProof) public virtual` — verifies `FHE.checkSignatures`, then `SafeERC20.safeTransfer(underlying, to, unwrapAmountCleartext * rate())`, emits `UnwrapFinalized`.
- Views: `decimals() → uint8` (min(underlyingDecimals, 6)); `rate() → uint256`; `underlying() → address`; `unwrapAmount(bytes32 unwrapRequestId) → euint64` (= `euint64.wrap(unwrapRequestId)`); `unwrapRequester(bytes32) → address` (0 if no pending request); `inferredTotalSupply() → uint256` (= `underlying.balanceOf(wrapper) / rate()`; inflatable by direct transfers; lags on unwrap); `maxTotalSupply() → uint256` (= `type(uint64).max`).
- Internal: `_maxDecimals() internal pure virtual returns (uint8)` → `6`; `_fallbackUnderlyingDecimals()` → `18`; `_checkConfidentialTotalSupply()` reverts `ERC7984TotalSupplyOverflow()` when `inferredTotalSupply() > maxTotalSupply()`; `_update` override runs that check on every mint (i.e., **wrap reverts on potential uint64 supply overflow** — CHANGELOG 0.4.0: "revert on wrap if there is a chance of total supply overflow").

### 6.2 Decimals / rate / rounding

Constructor logic (verbatim semantics): if `underlyingDecimals > 6`: wrapper `decimals = 6`, `rate = 10 ** (underlyingDecimals - 6)`; else `decimals = underlyingDecimals`, `rate = 1`.
- Zama doc table: 18-dec underlying → 6-dec wrapper, rate `10^12` ("1 wrapped = 10^12 underlying", e.g. WETH/cWETH); 6-dec (USDC/USDT) → rate 1, 1:1; 2-dec → 1:1.
- **Rounding:** wrap rounds DOWN to the nearest multiple of `rate`; `wrap()` simply doesn't pull the remainder; ERC-1363 path refunds it. Zama doc (verbatim): "If the amount is less than the rate, the wrapping will succeed but the recipient will receive 0 confidential tokens and the excess tokens will be refunded to the `msg.sender`."
- Unwrap pays out `cleartextAmount * rate()` underlying — no rounding loss on unwrap (wrapper units are the coarse unit).

### 6.3 Unwrap = async two-step (burn → public decryption → finalize)

`_unwrap` internal (verbatim key lines):
```solidity
require(to != address(0), ERC7984InvalidReceiver(to));
require(from == msg.sender || isOperator(from, msg.sender), ERC7984UnauthorizedSpender(from, msg.sender));

// try to burn, see how much we actually got
euint64 unwrapAmount_ = _burn(from, amount);
FHE.makePubliclyDecryptable(unwrapAmount_);
...
bytes32 unwrapRequestId = euint64.unwrap(unwrapAmount_);
_unwrapRequests[unwrapRequestId] = to;
emit UnwrapRequested(to, unwrapRequestId, unwrapAmount_);
```
- The request id IS the ciphertext handle of the **actually burned** amount (clamped to 0 if balance insufficient). Code WARNING (verbatim): "Directly using the cipher-text as the unwrap request id assumes that cipher-texts are unique--this holds here but is not always true. Be cautious when assuming cipher-text uniqueness."
- Tokens are burned at request time; **no underlying moves until `finalizeUnwrap`** ("**NO** transfer of underlying tokens is made in this request" — Zama doc). Anyone can call `finalizeUnwrap` once they hold the public-decryption proof (obtained via the relayer/gateway public decryption of the `UnwrapRequested` handle).
- SDK automates this: `wrappedToken.unshield(amount)` = unwrap → wait for decryption proof → `finalizeUnwrap`; also `unshieldAll()` (uses the balance handle directly, no proof needed for the amount), `unwrap(amount)`, `unwrapAll()`, `finalizeUnwrap(unwrapRequestId)`, `resumeUnshield(unwrapTxHash)`. (`_raw/erc7984/sdk-wrappedtoken.md`, `unshield-tokens.md`)
- Events (interface, verbatim): `event UnwrapRequested(address indexed receiver, bytes32 indexed unwrapRequestId, euint64 amount);` `event UnwrapFinalized(address indexed receiver, bytes32 indexed unwrapRequestId, euint64 encryptedAmount, uint64 cleartextAmount);` — note `UnwrapFinalized` reveals the CLEARTEXT amount on-chain.
- Interface note (verbatim): "The returned unwrap request id must never be zero."

## 9. Events & on-chain privacy

- Per transfer, observers see: `ConfidentialTransfer(address indexed from, address indexed to, euint64 indexed amount)` — `from`/`to` addresses are PUBLIC; `amount` is an indexed **ciphertext handle** (bytes32 topic), not a value. No amounts are decodable without ACL/decryption rights. Mints show `from = address(0)` (so wrap amounts correlate with the public `Wrap.roundedAmount` on deployed wrappers); burns show `to = address(0)`.
- `OperatorSet(holder, operator, until)` is fully public — reveals which pool a user authorized and until when.
- Cleartext leaks by design: deployed-wrapper `Wrap(to, roundedAmount, encryptedWrappedAmount)` (public rounded underlying amount), `UnwrapFinalized(..., cleartextAmount)` (public unwrap amount), `AmountDisclosed(encryptedAmount, amount)` (explicit disclosure).
- `_transferAndCall` privacy note (natspec verbatim): "Event observers see `sent` and `refund` individually" — two `ConfidentialTransfer` events are emitted (transfer + refund), so a callback deposit produces 2 events even when refund = 0.
- Indexing: SDK exposes `TOKEN_TOPICS`, `decodeOnChainEvents`, `decodeConfidentialTransfer`, `decodeWrap`, `decodeUnwrapRequested`, `decodeUnwrapFinalized`, `findWrap`, `findUnwrapRequested` (`_raw/erc7984/sdk-event-decoders.md`). Doc hint (verbatim): "A shield emits **both** a `ConfidentialTransfer(from=zeroAddress, …)` and a `Wrap` event. `Wrap.encryptedWrappedAmount` is the same FHE handle as the co-emitted `ConfidentialTransfer.encryptedAmount` — use `Wrap` as the shield marker (it carries the cleartext `roundedAmount`) and correlate the two halves rather than counting both."
- ACL delegation events (`DelegatedForUserDecryption`, `RevokedDelegationForUserDecryption`) come from the ACL contract, not the token; separate `ACL_TOPICS`.
